# ==========================================
# XMotor Industrial OPS 配置文件模板
# ==========================================
# 复制此文件为 .env 并根据实际情况修改配置
# 配置文件位置: %ProgramData%\XMotor-OPS\config\.env

# ==========================================
# 数据源配置
# ==========================================
# 是否使用 ModbusRTU 读取真实数据
# false = 使用模拟数据（用于测试和演示）
# true = 使用 ModbusRTU 读取真实设备数据
XMOTOR_USE_MODBUS=false

# ==========================================
# ModbusRTU 配置（仅在 USE_MODBUS=true 时生效）
# ==========================================
# 串口名称
# Windows: COM1, COM2, COM3, ... COM256
# 示例: COM5
XMOTOR_MODBUS_PORT=COM5

# 波特率（常见值: 9600, 19200, 38400, 115200）
XMOTOR_MODBUS_BAUDRATE=38400

# 从站地址（1-247）
XMOTOR_MODBUS_SLAVE_ID=1

# 串口参数
XMOTOR_MODBUS_PARITY=N
XMOTOR_MODBUS_STOPBITS=1
XMOTOR_MODBUS_BYTESIZE=8
XMOTOR_MODBUS_TIMEOUT=1.0

# 心跳间隔（秒），建议小于超时时间的一半
XMOTOR_HEARTBEAT_INTERVAL=0.5

# ==========================================
# 电机参数
# ==========================================
# 电机极对数（用于转速转换）
XMOTOR_MOTOR_POLE_PAIRS=4

# 转矩转换系数（转矩(Nm) = 电流(A) × 此系数）
# 默认 0.4 对应每安培 400 mN·m
XMOTOR_TORQUE_CURRENT_RATIO=0.4

# ==========================================
# 网络配置
# ==========================================
# CORS 允许的来源（多个用逗号分隔）
# ["*"] = 允许所有来源（开发环境）
# ["http://localhost", "http://192.168.1.100"] = 指定来源（生产环境）
XMOTOR_ALLOW_ORIGINS=["*"]

# ==========================================
# 日志配置
# ==========================================
# 日志级别: DEBUG, INFO, WARNING, ERROR
XMOTOR_LOG_LEVEL=INFO

# 是否输出日志到控制台
XMOTOR_LOG_TO_CONSOLE=true

# 日志文件路径（在容器内）
# 实际日志文件位置: %ProgramData%\XMotor-OPS\logs\
XMOTOR_LOG_FILE=logs/control_backend.log

# 日志文件最大大小（字节，默认 2MB）
XMOTOR_LOG_MAX_BYTES=2097152

# 日志备份文件数量
XMOTOR_LOG_BACKUP_COUNT=5
